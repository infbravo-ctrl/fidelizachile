        function animateCounters() {
            const statVisibilidad = document.getElementById('stat-visibilidad');
            const statClientes = document.getElementById('stat-clientes');
            const statMejora = document.getElementById('stat-mejora');
            const statPosicion = document.getElementById('stat-posicion');
            
            const metricImpresiones = document.getElementById('metric-impresiones');
            const metricRating = document.getElementById('metric-rating');
            const metricBusquedas = document.getElementById('metric-busquedas');
            const metricLlamadas = document.getElementById('metric-llamadas');
            
            const counters = [
                { element: statVisibilidad, target: 94, prefix: '+', suffix: '%' },
                { element: statClientes, target: 156, prefix: '+', suffix: '%' },
                { element: statMejora, target: 28, prefix: '+', suffix: '' },
                { element: statPosicion, target: 1, prefix: 'Nº ', suffix: '', isPosition: true },
                { element: metricImpresiones, target: 247, prefix: '+', suffix: '%' },
                { element: metricRating, target: 49, prefix: '', suffix: '★', divideBy10: true },
                { element: metricBusquedas, target: 183, prefix: '+', suffix: '%' },
                { element: metricLlamadas, target: 312, prefix: '+', suffix: '%' }
            ];
            
            counters.forEach(counter => {
                if (!counter.element) return;
                
                if (counter.isPosition) {
                    counter.element.textContent = 'Nº 1';
                    return;
                }
                
                if (counter.divideBy10) {
                    counter.element.textContent = counter.prefix + '0.0' + counter.suffix;
                } else {
                    counter.element.textContent = counter.prefix + '0' + counter.suffix;
                }
                
                const data = { value: 0 };
                
                gsap.to(data, {
                    value: counter.target,
                    duration: 2,
                    ease: "power2.out",
                    scrollTrigger: {
                        trigger: counter.element,
                        start: "top 90%",
                        toggleActions: "play none none reverse"
                    },
                    onUpdate: function() {
                        let currentValue;
                        
                        if (counter.divideBy10) {
                            currentValue = (data.value / 10).toFixed(1);
                        } else {
                            currentValue = Math.floor(data.value);
                        }
                        
                        counter.element.textContent = counter.prefix + currentValue + counter.suffix;
                    },
                    onComplete: function() {
                        let finalValue;
                        
                        if (counter.divideBy10) {
                            finalValue = (counter.target / 10).toFixed(1);
                        } else {
                            finalValue = counter.target;
                        }
                        
                        counter.element.textContent = counter.prefix + finalValue + counter.suffix;
                    }
                });
            });
        }

        setTimeout(animateCounters, 500);

                // Mostrar toast de éxito
                showToast('¡Mensaje enviado! Te contactaremos en menos de 2 horas.');
                
                // Resetear formulario
                form.reset();
                
            } catch (error) {
                console.error('Error:', error);
                
                // Fallback: abrir WhatsApp igual
                window.open(whatsappUrl, '_blank');
                
                // Mostrar toast de advertencia
                showToast('Redirigiendo a WhatsApp...', true);
                
                // Resetear formulario
                form.reset();
                
            } finally {
                // Restaurar botón
                submitBtn.innerHTML = originalText;
                submitBtn.disabled = false;
            }
        });

        // ===== FAQ ACCORDION =====
        document.querySelectorAll('.faq-question').forEach(question => {
            question.addEventListener('click', function() {
                const answer = this.nextElementSibling;
                const icon = this.querySelector('i');
                
                // Cerrar otros abiertos (opcional)
                document.querySelectorAll('.faq-answer.show').forEach(openAnswer => {
                    if (openAnswer !== answer) {
                        openAnswer.classList.remove('show');
                        openAnswer.previousElementSibling.classList.remove('active');
                        openAnswer.previousElementSibling.querySelector('i').classList.remove('fa-times');
                        openAnswer.previousElementSibling.querySelector('i').classList.add('fa-plus');
                    }
                });
                
                // Toggle actual
                answer.classList.toggle('show');
                this.classList.toggle('active');
                
                if (icon.classList.contains('fa-plus')) {
                    icon.classList.remove('fa-plus');
                    icon.classList.add('fa-times');
                } else {
                    icon.classList.remove('fa-times');
                    icon.classList.add('fa-plus');
                }
            });
        });

        // ===== SCROLL SUAVE =====
        document.querySelectorAll('a[href^="#"]').forEach(anchor => {
            anchor.addEventListener('click', function(e) {
                e.preventDefault();
                
                const targetId = this.getAttribute('href');
                if (targetId === '#') return;
                
                const targetElement = document.querySelector(targetId);
                if (targetElement) {
                    if (window.innerWidth <= 992) {
                        document.getElementById('navLinks').style.display = 'none';
                        document.getElementById('mobileMenuBtn').innerHTML = '<i class="fas fa-bars"></i>';
                    }
                    
                    window.scrollTo({
                        top: targetElement.offsetTop - 100,
                        behavior: 'smooth'
                    });
                }
            });
        });

        // ===== ACTUALIZAR AÑO =====
        const currentYear = new Date().getFullYear();
        document.getElementById('current-year').textContent = currentYear;
    </script>
